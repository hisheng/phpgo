# phpgo
用 go 实现 php 的array数组以及常用函数，给php转到go来的人用    


#install
go get github.com/hisheng/phpgo
    
    
        取值
        php 的 strval() 对应 phpgo.Strval()
        
        
        array
        php的array 数组结构 对应 phpgo.Array
        in_array()   phpgo.Array.In()
        array_pop()  phpgo.Array.Pop()
